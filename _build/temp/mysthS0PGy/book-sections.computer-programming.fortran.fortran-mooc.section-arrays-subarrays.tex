\subparagraph{Section Arrays: Subarrays}

Adapted from: \href{https://github.com/gjbex/Fortran-MOOC/tree/master/source\_code/arrays}{https://github.com/gjbex/Fortran-MOOC/tree/master/source\_code/arrays}

\subparagraph{This program demonstrates array slicing in Fortran.}

\begin{verbatim}
program subarrays
  implicit none
  integer :: i
  integer, dimension(10) :: A = [ (i, i = 1, 10) ]
  character(len=10), parameter :: FMT = '(10I3)'

  print FMT, A
  print FMT, A(4:7)
  print FMT, A(:7)
  print FMT, A(4:)
  print FMT, A(4:7:2)
  print FMT, A(4::2)
  print FMT, A(:7:2)
  print FMT, A(7:4: -1)
end program subarrays
\end{verbatim}
