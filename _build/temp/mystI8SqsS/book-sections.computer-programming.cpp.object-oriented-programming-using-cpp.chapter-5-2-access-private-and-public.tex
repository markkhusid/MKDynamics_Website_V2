\subparagraph{Access: Private and Public}

Adapted from: ``Object-Oriented Programming Using C++'' by Ira Pohl (Addison- Wesley)

\subparagraph{Program that demonstrates private and public access of object items in C++}

ch\_stac3.cpp

\begin{verbatim}
// Reverse a string with a ch_stack.
#include <iostream>
#include "ch_stac3.h"

using namespace std;

int main ()
{
    ch_stack s;
    char str[40] = {"My name is Don Knuth!"};
    int i = 0;

    cout << str << endl;
    s.reset();  // s.top = EMPTY would be illegal

    while (str[i] && !s.full())
    {
        s.push(str[i++]);
    }

    while (!s.empty())      // print the reverse
    {
        cout << s.pop();
    }

    cout << endl;
}
\end{verbatim}

ch\_stac3.h

\begin{verbatim}
struct ch_stack {
    public:
        void reset ()       { top = EMPTY; }
        void push (char c)  { top++; s[top] = c; }
        char pop ()         { return s[top - -]; }
        char top_of ()      { return s[top]; }
        bool empty ()       { return ( top == EMPTY ); }
        bool full ()        { return ( top == FULL  ); }

    private:
        enum 
        {  
            max_len = 100,
            EMPTY = -1,
            FULL = max_len -1
                
        };

        char s[max_len];
        int top;
};
\end{verbatim}
