\subparagraph{Section Call By Semantics: Call By In Reference}

Adapted from: \href{https://github.com/gjbex/Fortran-MOOC/tree/master/source\_code/call\_by\_semantics}{https://github.com/gjbex/Fortran-MOOC/tree/master/source\_code/call\_by\_semantics}

\subparagraph{This program demonstrates calling by in reference in Fortran.}

\begin{verbatim}
program call_by_in_reference
    implicit none

    print *, "5! = ", factorial(5)
    print *, "10! = ", factorial(10)
    print *, "12! = ", factorial(12)

contains

    function factorial(arg) result(fac)
        implicit none
        integer, intent(in) :: arg
        integer :: fac, n

        n = arg
        fac = 1
        do while (n > 1)
            fac = fac*n
            n = n - 1
        end do
    end function factorial

end program call_by_in_reference
\end{verbatim}
